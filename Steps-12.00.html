KPIs

1-      Sales KPIs

Total Sales Current Year = SUM(cali_avocados[TotalValue])

Sales Last Year = CALCULATE(SUM(cali_avocados[TotalValue]),DATEADD('Calandar Table'[Date],-1,YEAR))

Sales YoY% = DIVIDE(([Total Sales Current Year]-[.Sales Last Year]),[.Sales Last Year])

 

%YOY Sign = SWITCH(TRUE(),([%YOY])>0,UNICHAR(9650), ([%YOY])<0,UNICHAR(9660)," ")

 

%YOY Color = IF([%YOY]>0,"#1AAB40","#D6272C")

 

2-      Production KPIs

 

.YOY avg.Production current year = CALCULATE(AVERAGEX(cali_avocados, cali_avocados[ Production]))

 

.YOY avg.production last year = CALCULATE(AVERAGEX(cali_avocados, cali_avocados[ Production]),DATEADD('Calandar Table'[Date],-1,YEAR))

 

Production YoY% =

Var cy = [.YOY avg.Production current year]

var ly = [.YOY avg.production last year]

Return

DIVIDE((cy-ly), ly )

 

Sign for YOY avg.Production = SWITCH(TRUE(),([Production YoY%]*100)>0,UNICHAR(9650), ([Production YoY%]*100)<0,UNICHAR(9660)," ")

 

production YOY Color = IF([Production YoY%]>0,"#1AAB40","#D6272C")
